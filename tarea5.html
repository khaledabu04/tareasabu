Tarea #5

Encontrar un número del 1 al 1000

    #include <stdio.h> 
    #include <stdlib.h>

       int main()
       {
        int menor = 1, mayor = 1000, mitad;
        char respuesta;
    
        printf("Piensa en un número entre 0 y 1000\n Si lo adivino, me debes una gomita:)\n\n");
        printf("Responde con 'm' si es menor, 'M' si es mayor, o 'c' si es correcto\n\n");
    
        while (menor < mayor) {
            mitad = (menor + mayor) / 2;
            printf("Tu numero es: %d?\n", mitad);
            scanf(" %c", &respuesta);
    
            if (respuesta == 'c') {
                printf("Te lei la mente!:)\n");
                break;
            }

            else if (respuesta == 'm') {
                mayor = mitad - 1;  //
            }

            else if (respuesta == 'M') {
                menor = mitad + 1;
            }
        }
    
        return 0;
        }
